# Historique des commandes - TP Docker A1

# 1. Construction de l'image
docker build -t vanitas-flask-app .

# 2. Vérification de l'image
docker images

# 3. Lancement du conteneur
docker run -d -p 5000:5000 --name vanitas-flask vanitas-flask-app

# 4. Vérification du conteneur
docker ps

# 5. Tag de l'image pour DockerHub
docker tag vanitas-flask-app lionelrenaud1/vanitas-flask-app:v1

# 6. Push sur DockerHub
docker push lionelrenaud1/vanitas-flask-app:v1

# TP 2.3 - Docker Compose
docker compose up -d
docker compose ps

# TP 2.4 - Réseau Docker
docker network create custom_network
docker run -d --name cont1 --network custom_network busybox sleep 3600
docker run -d --name cont2 --network custom_network busybox sleep 3600
docker exec cont1 ping -c 2 cont2
docker network create net1
docker network create net2
docker run -d --name cont3 --network net1 busybox sleep 3600
docker run -d --name cont4 --network net2 busybox sleep 3600
docker exec cont3 ping -c 2 cont4
docker network connect net2 cont3
docker exec cont3 ping -c 2 cont4
